MainActivity.kt:

package com.example.myapplication

import android.Manifest
import android.app.NotificationChannel
import android.app.NotificationManager
import android.content.Context
import android.content.pm.PackageManager
import android.os.Build
import android.os.Bundle
import android.widget.Button
import android.widget.TextView
import android.widget.Toast
import androidx.activity.enableEdgeToEdge
import androidx.activity.result.contract.ActivityResultContracts
import androidx.appcompat.app.AppCompatActivity
import androidx.core.app.NotificationCompat
import androidx.core.app.NotificationManagerCompat
import androidx.core.content.ContextCompat
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat
import com.google.android.material.textfield.TextInputEditText
import com.google.firebase.firestore.FirebaseFirestore
import java.util.Date

class MainActivity : AppCompatActivity() {

    private var currentBalance: Double = 0.0
    private lateinit var tvBalance: TextView
    private lateinit var etAmount: TextInputEditText
    private lateinit var db: FirebaseFirestore

    private val CHANNEL_ID = "balance_alerts"
    private val NOTIFICATION_ID = 1

    private val requestPermissionLauncher = registerForActivityResult(
        ActivityResultContracts.RequestPermission()
    ) { isGranted: Boolean ->
        if (!isGranted) {
            Toast.makeText(this, "Notification permission denied. You won't receive alerts.", Toast.LENGTH_SHORT).show()
        }
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }

        db = FirebaseFirestore.getInstance()

        tvBalance = findViewById(R.id.tvBalance)
        etAmount = findViewById(R.id.etAmount)
        val btnAddSalary: Button = findViewById(R.id.btnAddSalary)
        val btnAddExpense: Button = findViewById(R.id.btnAddExpense)

        createNotificationChannel()
        checkNotificationPermission()
        
        // Optionally load latest balance from Firestore if needed
        // For now, we'll just focus on saving records as requested.

        btnAddSalary.setOnClickListener {
            val amount = etAmount.text.toString().toDoubleOrNull()
            if (amount != null) {
                currentBalance += amount
                updateBalanceUI()
                saveRecordToFirebase("salary", amount)
                etAmount.text?.clear()
            } else {
                Toast.makeText(this, "Please enter a valid amount", Toast.LENGTH_SHORT).show()
            }
        }

        btnAddExpense.setOnClickListener {
            val amount = etAmount.text.toString().toDoubleOrNull()
            if (amount != null) {
                currentBalance -= amount
                updateBalanceUI()
                saveRecordToFirebase("expense", amount)
                checkBalanceAndNotify()
                etAmount.text?.clear()
            } else {
                Toast.makeText(this, "Please enter a valid amount", Toast.LENGTH_SHORT).show()
            }
        }
    }

    private fun saveRecordToFirebase(type: String, amount: Double) {
        val record = hashMapOf(
            "type" to type,
            "amount" to amount,
            "timestamp" to Date(),
            "resultingBalance" to currentBalance
        )

        db.collection("records")
            .add(record)
            .addOnSuccessListener {
                Toast.makeText(this, "Record saved to Firebase", Toast.LENGTH_SHORT).show()
            }
            .addOnFailureListener { e ->
                Toast.makeText(this, "Error saving record: ${e.message}", Toast.LENGTH_LONG).show()
            }
    }

    private fun updateBalanceUI() {
        tvBalance.text = String.format("%.2f", currentBalance)
    }

    private fun checkBalanceAndNotify() {
        if (currentBalance < 1000) {
            sendLowBalanceNotification()
        }
    }

    private fun createNotificationChannel() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.O) {
            val name = "Balance Alerts"
            val descriptionText = "Notifications when balance goes below 1000"
            val importance = NotificationManager.IMPORTANCE_DEFAULT
            val channel = NotificationChannel(CHANNEL_ID, name, importance).apply {
                description = descriptionText
            }
            val notificationManager: NotificationManager =
                getSystemService(Context.NOTIFICATION_SERVICE) as NotificationManager
            notificationManager.createNotificationChannel(channel)
        }
    }

    private fun checkNotificationPermission() {
        if (Build.VERSION.SDK_INT >= Build.VERSION_CODES.TIRAMISU) {
            if (ContextCompat.checkSelfPermission(this, Manifest.permission.POST_NOTIFICATIONS) != PackageManager.PERMISSION_GRANTED) {
                requestPermissionLauncher.launch(Manifest.permission.POST_NOTIFICATIONS)
            }
        }
    }

    private fun sendLowBalanceNotification() {
        val builder = NotificationCompat.Builder(this, CHANNEL_ID)
            .setSmallIcon(android.R.drawable.ic_dialog_alert)
            .setContentTitle("Low Balance Alert")
            .setContentText("Your balance is below 1000! Current balance: ${String.format("%.2f", currentBalance)}")
            .setPriority(NotificationCompat.PRIORITY_DEFAULT)

        with(NotificationManagerCompat.from(this)) {
            if (ContextCompat.checkSelfPermission(this@MainActivity, Manifest.permission.POST_NOTIFICATIONS) == PackageManager.PERMISSION_GRANTED || Build.VERSION.SDK_INT < Build.VERSION_CODES.TIRAMISU) {
                notify(NOTIFICATION_ID, builder.build())
            }
        }
    }
}
