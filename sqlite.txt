MainActivity.kt:

package com.example.not

import android.os.Bundle
import android.widget.Button
import android.widget.EditText
import android.widget.Toast
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat
import androidx.recyclerview.widget.LinearLayoutManager
import androidx.recyclerview.widget.RecyclerView

class MainActivity : AppCompatActivity() {

    private lateinit var dbHelper: DatabaseHelper
    private lateinit var adapter: NotesAdapter
    private lateinit var etTitle: EditText
    private lateinit var etContent: EditText
    private lateinit var btnSave: Button
    private var editingNoteId: Int? = null

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }

        dbHelper = DatabaseHelper(this)
        etTitle = findViewById(R.id.etTitle)
        etContent = findViewById(R.id.etContent)
        btnSave = findViewById(R.id.btnSave)
        val rvNotes = findViewById<RecyclerView>(R.id.rvNotes)

        adapter = NotesAdapter(
            notes = dbHelper.getAllNotes(),
            onEditClick = { note ->
                etTitle.setText(note.title)
                etContent.setText(note.content)
                editingNoteId = note.id
                btnSave.text = "Update Note"
            },
            onDeleteClick = { note ->
                dbHelper.deleteNote(note.id)
                refreshNotes()
                Toast.makeText(this, "Note deleted", Toast.LENGTH_SHORT).show()
            }
        )

        rvNotes.layoutManager = LinearLayoutManager(this)
        rvNotes.adapter = adapter

        btnSave.setOnClickListener {
            val title = etTitle.text.toString()
            val content = etContent.text.toString()

            if (title.isNotEmpty() && content.isNotEmpty()) {
                if (editingNoteId == null) {
                    dbHelper.addNote(title, content)
                    Toast.makeText(this, "Note saved", Toast.LENGTH_SHORT).show()
                } else {
                    dbHelper.updateNote(editingNoteId!!, title, content)
                    Toast.makeText(this, "Note updated", Toast.LENGTH_SHORT).show()
                    editingNoteId = null
                    btnSave.text = "Save Note"
                }
                clearInputs()
                refreshNotes()
            } else {
                Toast.makeText(this, "Please enter title and content", Toast.LENGTH_SHORT).show()
            }
        }
    }

    private fun refreshNotes() {
        adapter.updateNotes(dbHelper.getAllNotes())
    }

    private fun clearInputs() {
        etTitle.text.clear()
        etContent.text.clear()
    }
}
